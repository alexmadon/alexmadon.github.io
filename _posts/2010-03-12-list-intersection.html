scala> val a=List(1,2,3,4,5,10,11,12,13,14,15) 
a: List[Int] = List(1, 2, 3, 4, 5, 10, 11, 12, 13, 14, 15)

scala> val b=List(8,9,10,11,12)
b: List[Int] = List(8, 9, 10, 11, 12)
             ^

scala> b.intersect(a)
res1: List[Int] = List(10, 11, 12)
