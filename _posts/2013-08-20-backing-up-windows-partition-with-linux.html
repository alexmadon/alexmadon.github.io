clone /dev/hda1 to hda1.img and display debug information.
partclone.ntfs -c -d -s /dev/hda1 -o hda1.img

restore /dev/hda1 from hda1.img and display debug information.
partclone.ntfs -r -d -s hda1.img -o /dev/hda1
