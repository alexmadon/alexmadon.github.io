cat video1.mpg video2.mpg | ffmpeg -y -i - -fflags genpts -vcodec copy -acodec copy video1+2.mpg
See http://lists.mplayerhq.hu/pipermail/ffmpeg-user/2006-October/004619.html For wmv files: 
for file in *wmv
do
    mencoder -idx $file -ovc copy -ovc lavc -lavcopts vcodec=msmpeg4v2 -o tmp/$file.mp4
done

mencoder -noidx -ovc copy -oac copy -o total.mp4 tmp/*mp4
All video files must have identical fps, resolution, and codec for -ovc copy.
