scala> val a=List(1,2,3,4,5,6) 
a: List[Int] = List(1, 2, 3, 4, 5, 6)

scala> def f(i:Int):Boolean = {(i%2)==0}
f: (Int)Boolean

scala> a filter f
res4: List[Int] = List(2, 4, 6)

scala> a.filter(f)
res5: List[Int] = List(2, 4, 6)
